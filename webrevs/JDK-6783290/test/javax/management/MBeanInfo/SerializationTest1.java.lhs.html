  89     public static void test(Object obj) throws Exception {
  90         ByteArrayOutputStream baos = new ByteArrayOutputStream();
  91         ObjectOutputStream oos = new ObjectOutputStream(baos);
  92         oos.writeObject(obj);
  93 

  94         ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
  95         ObjectInputStream ois = new ObjectInputStream(bais);













  96         Object newObj = ois.readObject();
  97 
  98         if (!obj.equals(newObj)) {
  99             throw new RuntimeException("Serialization/deserialization failed.");
 100         }
